	\begin{frame}{Problem}
		
		Time series classification originated from the $K$ dynamical systems of the form
		
		 \begin{equation}\label{eq:lat_dyn}
		 	\begin{cases}
		 		\frac{d}{dt} \bz(t) = f(\bz), \\
		 		\bz(0) = \bz_0,
		 	\end{cases}
		 \end{equation}
		 
		assuming we have labeled sample series $\mathcal{D} = \{ (\tbx_1(t_i), y_1), \ldots, (\tbx_N(t_i), y_N) \}$.
		
		\begin{figure}[h]
			\includegraphics[width=0.49\textwidth, keepaspectratio]{img/dws_example}
			\includegraphics[width=0.49\textwidth, keepaspectratio]{img/jog_example}
			\includegraphics[width=0.49\textwidth, keepaspectratio]{img/stand_example}
			\caption*{Time series from different systems}
		\end{figure}
		
	\end{frame}% say tilde means noised
	
	\begin{frame}{"Direct" approach}
		
		\begin{enumerate}
			\item Reconstruct diffeomorphic phase trajectories from the observed time series: Whitney or Takens embedding theorem.
			\begin{equation*}
				\{\tbx(t_i)\} \to \{\bz(t_i)\}
			\end{equation*}
			\item Learn vector fields of the systems $f_i(\bz)$ from the trajectories (NODE).
			\begin{equation*}
				\hat{f} = \underset{f \in \mathcal{F}}{\arg\min} \  \mathcal{L}(\{\bz(t_i)\}, f)
			\end{equation*}
			
			Choose $\mathcal{L}, \mathcal{F}$ according to the prior knowledge of the system.			
			\item Associate a test trajectory with the system that gives least deviation from the \emph{initial value problem} (IVP) (\ref{eq:lat_dyn}).
			
			\begin{equation*}
				k = \underset{k \in 1 \ldots K}{\arg\min} \  \rho \big( \{\bz(t_i)\}, \text{IVP}(f_k, \bz(0)) \big)
			\end{equation*}
		\end{enumerate}
		
	\end{frame}
	
	\begin{frame}{Approach limitations}
		
		In steps 2-3 we need to compute IVP trajectories. For regular fields $f$ and fixed $\bz(0)$ IVP has unique solution. On practice, there are several factors that induce error in computing $\text{IVP}(f, \bz(0))$.
		
		\begin{enumerate}
			\item In many cases IVP has no analytic solution and \emph{numerical solvers} are used. They induce numerical error and may have stability issues.
			\item Learned field $\hat{f}$ from step 2 is an approximation of the real $f$. It contributes to the IVP error.
			\item We have only noisy $\bz(0)$ observation.
			\item The dynamical system (\ref{eq:lat_dyn}) may not be an adequate time series model globally. For example, the true field is not stationary $f(\bz, t)$. But it can fairly approximates dynamics locally.
		\end{enumerate}
		
	\end{frame}
	
	\begin{frame}{Trajectory segmenting}
		
		Do not rely for $\text{IVP}(f, \bz(0))$. We know full (noisy) trajectory $\{\bz(t_i)\}$. Split it into parts and solve IVP per part. We can also add parameters for $\bz(0)$ if noise is significant
		
		\begin{equation*}
			k = \underset{k \in 1 \ldots K}{\arg\min} \sum\limits_{s=0}^S  \rho \big( \{\bz^s(t_i)\}, \text{IVP}(f_k, \bz^s_0) \big)
		\end{equation*}
		
		In terms of \emph{bias-variance tradeoff} adding $\bz_0$ parameters decrease bias but will increase variance. Trajectory segmenting also allows for \emph{parallel} trajectory processing.
		
		\vspace{-1.5em}		
		\begin{figure}[h]
			\subfloat[No segmenting]{\includegraphics[width=0.49\textwidth, keepaspectratio]{img/no_segmenting}}
			\subfloat[With segmenting]{\includegraphics[width=0.49\textwidth, keepaspectratio]{img/segmenting}}
		\end{figure}
		
	\end{frame}
	
	\begin{frame}{"Transition to model space" approach}
		
		Instead of analyzing trajectories let's analyze vector field $f$ directly. 
		
		\begin{enumerate}
			\item Split trajectory on the \emph{segments} (with possibly adaptive sizes).
			\item Assume linear field approximation is fair inside the segments
				\begin{equation*}
					\dot{x} = f(\bx) \approx f(\bx_0) + \left.\mathcal{I}\right|_{\bx_0} (\bx - \bx_0)
				\end{equation*}
			and learn it (i.e. $f(\bx_0)$ and $\left.\mathcal{I}\right|_{\bx_0}$).
			\item Do it for every train and test trajectory. In the end we have samples of $f$ values across the phase space.
		\end{enumerate}
		
		\emph{One possible} solution is then:
		
		\begin{enumerate}
			\item Build global parametric approximation $\hat{f}_k, \ k \in 1 \ldots K$.
			\item Classifier is the system with the minimum deviation of the vector field samples of the test trajectory
				\begin{equation*}
					k = \underset{k \in 1 \ldots K}{\arg\min} \  \rho \big( \{f(t_i)\}, \{\hat{f}_k(t_i)\} \big)
				\end{equation*}
		\end{enumerate}
		
	\end{frame}
